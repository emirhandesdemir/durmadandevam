rules_version = '2';

service firebase.storage {
  match /b/{bucket}/o {
    // Allow public read on all files. This is common for profile pictures, post images, etc.
    match /{allPaths=**} {
      allow read;
    }
    
    // Allow an authenticated user to write to their own avatar folder.
    // Users can only upload one avatar, and it must be a small image file.
    match /avatars/{userId} {
      allow write: if request.auth != null && request.auth.uid == userId 
                     && request.resource.size < 5 * 1024 * 1024
                     && request.resource.contentType.matches('image/.*');
    }

    // Allow an authenticated user to write to their own posts folder.
    // Post images must be small image files.
    match /posts/{userId}/{allPaths=**} {
       allow write: if request.auth != null && request.auth.uid == userId
                     && request.resource.size < 10 * 1024 * 1024
                     && request.resource.contentType.matches('image/.*');
    }
  }
}
